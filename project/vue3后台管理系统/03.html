# Koa2项目初始化操作

# koa-generator 快速生成服务

  1. 全局安装脚手架工具
npm i koa-generator -g
  1. 创建项目
# manager-server 是我创建的项目名，可以根据实际情况替换
koa2 manager-server
  1. 安装依赖
npm install
  1. 启动服务
npm start

# 使用 pm2 部署 koa 项目

  1. 全局安装
npm install pm2 -g
  1. 启动项目
# app.js 是项目入口文件名
pm2 start app.js
  1. pm2 自动重启

把 pm2 的服务停止后，然后起来的时候带上 --watch
  1. 启动完成

# pm2 相关命令

pm2 list # 查看所有已启动项目
pm2 start # 启动
pm2 restart www # 重启
pm2 stop www # 停止
pm2 delete www# 删除

# 应用log4js进行日志规范封装

具体使用方法，请详细查阅 log4js官网 (opens new window) 。这里仅简要说下：

安装

npm i log4js -S

简单使用：

var log4js = require("log4js");
var logger = log4js.getLogger();
logger.level = "debug";
logger.debug("Some debug messages");

level 默认是关闭的，不输出日志，如果需要输出日志，需要设置 level 。

level 是分级别的，优先级由上往下，比如我们上面设置的 level 为 debug，就只能输出 debug 和 debug 下面的级别。 级别大体为：

trace
debug
info
error
warn
fatal

来看一个例子：

var log4js = require("log4js");
var logger = log4js.getLogger();
logger.level = "debug";
logger.trace("trace");
logger.debug("debug");
logger.info("info");
logger.error("error");
logger.warn("warn");
logger.fatal("fatal");

可以看到没有输出 'trace' 。

但是这个顺序可是不固定的，比如我们把 fatal 放到 debug 上面，也会先输出 fatal。但是把 trace 放到最下面，依然不会输出 trace 。

var log4js = require("log4js");
var logger = log4js.getLogger();
logger.level = "debug";
logger.fatal("fatal");
logger.debug("debug");
logger.info("info");
logger.error("error");
logger.warn("warn");
logger.trace("trace");

可以看到先输出了 fatal 。

封装一个 log.js 文件，用来处理日志输出：

const log4js = require('log4js');

const levels = {
	trace: log4js.levels.TRACE,
	debug: log4js.levels.DEBUG,
	info: log4js.levels.INFO,
	error: log4js.levels.ERROR,
	warn: log4js.levels.WARN,
	fatal: log4js.levels.FATAL
};

log4js.configure({
	appenders: {
		console: { type: 'console' },
		info: {
			type: 'file',
			filename: 'log/all-logs.log'
		},
		error: {
			type: 'dateFile',
			filename: 'log/log',
			pattern: 'yyyy-MM-dd.log',
			alwaysIncludePattern: true
		}
	},
	categories: {
		default: { appenders: [ 'console' ], level: 'debug' },
		info: {
			appenders: [ 'info', 'console' ],
			level: 'info'
		},
		error: { appenders: [ 'error', 'console' ], level: 'error' }
	}
});

exports.debug = (content) => {
	const logger = log4js.getLogger();
	logger.level = levels.debug;
	logger.debug(content);
};

exports.info = (content) => {
	const logger = log4js.getLogger('info');
	logger.level = levels.info;
	logger.info(content);
};

exports.error = (content) => {
	const logger = log4js.getLogger('error');
	logger.level = levels.error;
	logger.error(content);
};

下面我们来使用下我们封装的 log.js 文件。在 app.js 文件中导入的 log.js 文件：

// ...
const log4js = require('./utils/log')
// ...

// logger
app.use(async (ctx, next) => {
  await next()
  log4js.info('log output');

  // 人为制造报错
  console.log(dddd);
})
// ...
app.on('error', (err, ctx) => {
  log4js.error(err.stack);
});

module.exports = app

# Mac下Mongo安装和配置

打开MongoDB 社区版下载地址 (opens new window)选择 MongoDB Community Server ，然后点击下载。（由于本人电脑系统版本较低，所以没有选择安装最新版本，我安装的版本是 3.6.23）

将下载后的安装包进行解压，将加压后的文件改名，如改为： mongoDB，然后将这个包放到想要运行 mongoDB 的位置。我这里将它复制到了电脑上的 Applications 文件中。

配置环境变量

export PATH=/Applications/mongoDB/bin:$PATH

在终端输入：

mongod -version

如果出现如下信息，则说明安装成功。

更详细的安装步骤请阅读 官方安装说明 (opens new window)。

在 mongoDB 目录新建 /data/db 目录和 /log/mongo.log 文件以及配置文件 mongo.conf 并输入如下代码：

#数据库路径
dbpath=/Applications/mongoDB/data/db
#日志输出路径
logpath=/Applications/mongoDB/log/mongo.log
#错误日志采用追加模式
logappend=true
#启用日志文件
journal=true
#过滤无用日志信息
quiet=true
#端口号
port=27017
#auth=true
fork=false

在终端执行如下命令：

mongod -f /Applications/mongoDB/mongo.conf

执行完命令后，当终端没有报错，且卡主后，新打开一个终端，进入到 mongoDB 的 bin 目录，执行：

./mongo

出现下面界面则说明连接成功了：

which 命令的作用是，在 PATH 变量指定的路径中，搜索某个系统命令的位置，并且返回第一个搜索结果。也就是说，使用 which 命令，就可以看到某个系统命令是否存在，以及执行的到底是哪一个位置的命令。

which mongod

mkdir -p ：递归创建目录，即使上级目录不存在，会按目录层级自动创建目录

# Mongo基础入门

SQL Mongo
表(Table) 集合(collection)
行(Row) 文档(Document)
列(Col) 字段(Field)
主键(Primary key) 对象ID(ObjectId)

数据库操作

操作 命令
创建数据库 use xxx
查看数据库 show dbs
删除数据库 db.dropDatabase()

集合操作

操作 命令
创建集合 db.createCollection(name)
查看集合 show collections
删除集合 db.xxx.drop()

文档操作

操作 命令
创建文档 db.xxx.insertOne()、db.xxx.insertMany([])
查看文档 db.xxx.find()
删除文档 db.xxx.deleteOne、db.xxx.deleteMany()
更新文档 db.xxx.update({},{},false, true)

条件操作

操作 命令
大于 $gt
小于 $lt
大于等于 $gte
小于等于 $lte

举例：

show dbs
use demo
db.createCollection("users")
show collections
db.users.insertOne({})
db.users.insertMany([])
db.users.find()
db.users.deleteOne({userId:3})
db.users.find({age:{$gt:35}})
db.users.update({age:{$get:30}}, {$set:{score:600}}, false, true)

# 工具函数封装

utils/utils.js

const CODE = {
  SUCCESS: 200, // 成功
  PARAM_ERROR: 10001, // 参数错误
  USER_ACCOUNT_ERROR: 20001, // 账号或密码错误
  USER_LOGIN_ERROR: 30001, // 用户未登录
  BUSINESS_ERROR: 40001, // 业务请求失败
  AUTH_ERROR: 50001 // 认证失败或 token 过期
};

module.exports = {
  parge({ pageNum=1, pageSize=10 }) {
    pageNum *= 1;
    pageSize *= 1;
    const skipIndex = (pageNum - 1) * pageSize;
    return {
      page: {
        pageNum,
        pageSize
      },
      skipIndex
    };
  },
  success(data, msg, code=CODE.SUCCESS) {
    return {
      code, data, msg
    };
  },
  fail(msg='', code=BUSINESS_ERROR) {
    return {
      msg, code
    }
  }
};
